  26 package hellotest;
  27 
  28 import javafx.application.Application;
  29 import javafx.event.ActionEvent;
  30 import javafx.scene.Group;
  31 import javafx.scene.Scene;
  32 import javafx.scene.control.Button;
  33 import javafx.stage.Stage;
  34 import com.sun.glass.events.KeyEvent;
  35 import com.sun.glass.ui.Robot;
  36 
  37 public class RemoveFocusedControl extends Application {
  38 
  39     public static void main(String[] args) {
  40         Application.launch(args);
  41     }
  42 
  43     @Override
  44     public void start(Stage stage) {
  45 
  46         final Group root = new Group();
  47         stage.setTitle("Remove Focused Control Test");
  48         Scene scene = new Scene(root, 600, 450);
  49 
  50         scene.setOnKeyTyped((javafx.scene.input.KeyEvent e) -> {
  51             System.out.println("Event: " + e);
  52         });
  53 
  54         Button button1 = new Button();
  55         button1.setText("Click HERE for removing this button and then sending "
  56                 + "four key events");
  57         button1.setLayoutX(25);
  58         button1.setLayoutY(40);
  59 
  60         final Robot robot = com.sun.glass.ui.Application.GetApplication().
  61                 createRobot();
  62 
  63         button1.setOnAction((ActionEvent e) -> {
  64             root.getChildren().remove(button1);
  65             robot.keyPress(KeyEvent.VK_T);
  66             robot.keyRelease(KeyEvent.VK_T);
  67             robot.keyPress(KeyEvent.VK_E);
  68             robot.keyRelease(KeyEvent.VK_E);
  69             robot.keyPress(KeyEvent.VK_S);
  70             robot.keyRelease(KeyEvent.VK_S);
  71             robot.keyPress(KeyEvent.VK_T);
  72             robot.keyRelease(KeyEvent.VK_T);
  73         });
  74 
  75         root.getChildren().add(button1);
  76 
  77         stage.setScene(scene);
  78         stage.show();
  79         System.out.println("Please press on 'Click HERE...' button in order to remove\n"
  80                 + " the button and generate automatically four keyEvents.\n"
  81                 + "Expected results: Button is disappearing, exactly FOUR KeyEvents\n"
  82                 + " are being accepted and printed to the console.");
  83     }
  84 }
