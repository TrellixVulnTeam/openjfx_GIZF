  25 package com.sun.glass.ui.mac;
  26 
  27 import com.sun.glass.ui.*;
  28 
  29 /**
  30  * MacOSX platform implementation class for Robot.
  31  */
  32 final class MacRobot extends Robot {
  33 
  34     // TODO: get rid of native Robot object
  35     private long ptr;
  36 
  37     private native long _init();
  38     @Override protected void _create() {
  39         ptr = _init();
  40     }
  41 
  42     private native void _destroy(long ptr);
  43     @Override protected void _destroy() {
  44         if (ptr == 0) {
  45             return;
  46         }
  47         _destroy(ptr);
  48     }
  49 
  50     @Override native protected void _keyPress(int code);
  51     @Override native protected void _keyRelease(int code);
  52 
  53     private native void _mouseMove(long ptr, int x, int y);
  54     @Override protected void _mouseMove(int x, int y) {
  55         if (ptr == 0) {
  56             return;
  57         }
  58         _mouseMove(ptr, x, y);
  59     }
  60 
  61     private native void _mousePress(long ptr, int buttons);
  62     @Override protected void _mousePress(int buttons) {
  63         if (ptr == 0) {
  64             return;
  65         }
  66         _mousePress(ptr, buttons);
  67     }
  68 
  69     private native void _mouseRelease(long ptr, int buttons);
  70     @Override protected void _mouseRelease(int buttons) {
  71         if (ptr == 0) {
  72             return;
  73         }
  74         _mouseRelease(ptr, buttons);
  75     }
  76 
  77     @Override native protected void _mouseWheel(int wheelAmt);
  78 
  79     private native int _getMouseX(long ptr);
  80     @Override protected int _getMouseX() {
  81         if (ptr == 0) {
  82             return 0;
  83         }
  84         return _getMouseX(ptr);
  85     }
  86 
  87     private native int _getMouseY(long ptr);
  88     @Override protected int _getMouseY() {
  89         if (ptr == 0) {
  90             return 0;
  91         }
  92         return _getMouseY(ptr);
  93     }
  94 
  95     @Override native protected int _getPixelColor(int x, int y);
  96     @Override native protected Pixels _getScreenCapture(int x, int y, int width, int height, boolean isHiDPI);
  97 }
