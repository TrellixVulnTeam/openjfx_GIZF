  25 package com.sun.glass.ui.win;
  26 
  27 import javafx.scene.input.KeyCode;
  28 import javafx.scene.input.MouseButton;
  29 import javafx.scene.paint.Color;
  30 import javafx.scene.robot.Robot;
  31 
  32 import com.sun.glass.ui.Application;
  33 import com.sun.glass.ui.GlassRobot;
  34 
  35 /**
  36  * MS Windows platform implementation class for Robot.
  37  */
  38 final class WinRobot extends Robot {
  39 
  40     @Override protected void create() {
  41         // no-op
  42     }
  43     @Override protected void destroy() {
  44         // no-op
  45     }
  46 
  47     native protected void _keyPress(int code);
  48     @Override public void keyPress(KeyCode code) {
  49         Application.checkEventThread();
  50         _keyPress(code.getCode());
  51     }
  52 
  53     native protected void _keyRelease(int code);
  54     @Override public void keyRelease(KeyCode code) {
  55         Application.checkEventThread();
  56         _keyRelease(code.getCode());
  57     }
  58 
  59 
  60     native protected void _mouseMove(int x, int y);
  61     @Override public void mouseMove(int x, int y) {
  62         Application.checkEventThread();
  63         _mouseMove(x, y);
  64     }
  65 
  66     native protected void _mousePress(int buttons);
  67     @Override public void mousePress(MouseButton button) {
  68         Application.checkEventThread();
  69         _mousePress(GlassRobot.convertToRobotMouseButton(button));
  70     }
  71 
  72     @Override
  73     public void mousePress(MouseButton... buttons) {
  74         Application.checkEventThread();
  75         _mousePress(GlassRobot.convertToRobotMouseButton(buttons));
  76     }
  77 
  78     native protected void _mouseRelease(int buttons);
  79     @Override public void mouseRelease(MouseButton button) {
  80         Application.checkEventThread();
  81         _mouseRelease(GlassRobot.convertToRobotMouseButton(button));
  82     }
  83 
  84     @Override
  85     public void mouseRelease(MouseButton... buttons) {
  86         Application.checkEventThread();
  87         _mouseRelease(GlassRobot.convertToRobotMouseButton(buttons));
  88     }
  89 
  90     native protected void _mouseWheel(int wheelAmt);
  91     @Override protected void mouseWheel(int wheelAmt) {
  92         Application.checkEventThread();
  93         _mouseWheel(wheelAmt);
  94     }
  95 
  96     native protected int _getMouseX();
  97     @Override public int getMouseX() {
  98         Application.checkEventThread();
  99         return _getMouseX();
 100     }
 101 
 102     native protected int _getMouseY();
 103     @Override public int getMouseY() {
 104         Application.checkEventThread();
 105         return _getMouseY();
 106     }
 107 
 108     native protected int _getPixelColor(int x, int y);
 109     @Override public Color getPixelColor(int x, int y) {
 110         Application.checkEventThread();
 111         return GlassRobot.convertFromIntArgb(_getPixelColor(x, y));
 112     }
 113 
 114     @Override native protected void getScreenCapture(int x, int y, int width, int height, int[] data);
 115 }
