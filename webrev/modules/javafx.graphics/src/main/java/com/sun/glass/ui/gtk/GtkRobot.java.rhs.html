 119     @Override
 120     public Color getPixelColor(int x, int y) {
 121         Application.checkEventThread();
 122         Screen mainScreen = Screen.getMainScreen();
 123         x = (int) Math.floor((x + 0.5) * mainScreen.getPlatformScaleX());
 124         y = (int) Math.floor((y + 0.5) * mainScreen.getPlatformScaleY());
 125         int[] result = new int[1];
 126         getScreenCapture(x, y, 1, 1, result);
 127         return GlassRobot.convertFromIntArgb(result[0]);
 128     }
