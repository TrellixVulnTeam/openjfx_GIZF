  26 package com.sun.glass.ui.ios;
  27 
  28 import com.sun.glass.ui.*;
  29 import java.nio.IntBuffer;
  30 
  31 /**
  32  * iOS platform implementation class of test automation Robot.
  33  */
  34 final class IosRobot extends Robot {
  35 
  36     private long ptr = 0;
  37 
  38     // init and create native robot object
  39     private native long _init();
  40     @Override protected void _create() {
  41         ptr = _init();
  42     }
  43 
  44     // release native robot object
  45     private native void _destroy(long ptr);
  46     @Override protected void _destroy() {
  47         _destroy(ptr);
  48         ptr = 0;
  49     }
  50 
  51     // synthesize key press
  52     private native void _keyPress(long ptr, int code);
  53     @Override protected void _keyPress(int code) {
  54         if (ptr == 0) {
  55             return;
  56         }
  57         _keyPress(ptr, code);
  58     }
  59 
  60     // synthesize key release
  61     private native void _keyRelease(long ptr, int code);
  62     @Override protected void _keyRelease(int code) {
  63         if (ptr == 0) {
  64             return;
  65         }
  66         _keyRelease(ptr, code);
  67     }
  68 
  69     // synthesize mouse motion
  70     private native void _mouseMove(long ptr, int x, int y);
  71     @Override protected void _mouseMove(int x, int y) {
  72         if (ptr == 0) {
  73             return;
  74         }
  75         _mouseMove(ptr, x, y);
  76     }
  77 
  78     // synthesize mouse press of buttons
  79     private native void _mousePress(long ptr, int buttons);
  80     @Override protected void _mousePress(int buttons) {
  81         if (ptr == 0) {
  82             return;
  83         }
  84         _mousePress(ptr, buttons);
  85     }
  86 
  87     // synthesize mouse release of buttons
  88     private native void _mouseRelease(long ptr, int buttons);
  89     @Override protected void _mouseRelease(int buttons) {
  90         if (ptr == 0) {
  91             return;
  92         }
  93         _mouseRelease(ptr, buttons);
  94     }
  95 
  96     private native void _mouseWheel(long ptr, int wheelAmt);
  97     @Override protected void _mouseWheel(int wheelAmt) {
  98         if (ptr == 0) {
  99             return;
 100         }
 101         _mouseWheel(ptr, wheelAmt);
 102     }
 103 
 104     // get x-coordinate of mouse location
 105     private native int _getMouseX(long ptr);
 106     @Override protected int _getMouseX() {
 107         if (ptr == 0) {
 108             return 0;
 109         }
 110         return _getMouseX(ptr);
 111     }
 112 
 113     // get x-coordinate of mouse location
 114     private native int _getMouseY(long ptr);
 115     @Override protected int _getMouseY() {
 116         if (ptr == 0) {
 117             return 0;
 118         }
 119         return _getMouseY(ptr);
 120     }
 121 
 122     private native int _getPixelColor(long ptr, int x, int y);
 123     @Override protected int _getPixelColor(int x, int y) {
 124         if (ptr == 0) {
 125             return 0;
 126         }
 127         return _getPixelColor(ptr, x, y);
 128     }
 129 
 130     // capture bitmap image of (x, y, x + width, y + height) area
 131     native private void _getScreenCapture(long ptr, int x, int y, int width, int height, int[] data);
 132     @Override protected Pixels _getScreenCapture(int x, int y, int width, int height, boolean isHiDPI) {
 133         if (ptr == 0) {
 134             return null;
 135         }
 136         int data[] = new int[width * height];
 137         _getScreenCapture(ptr, x, y, width, height, data);
 138         return Application.GetApplication().createPixels(width, height, IntBuffer.wrap(data));
 139     }
 140 }
