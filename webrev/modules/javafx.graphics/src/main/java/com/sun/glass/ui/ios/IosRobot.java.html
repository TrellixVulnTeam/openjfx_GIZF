  26 package com.sun.glass.ui.ios;
  27 
  28 import com.sun.glass.ui.Application;
  29 import com.sun.glass.ui.GlassRobot;
  30 
  31 import javafx.scene.input.KeyCode;
  32 import javafx.scene.input.MouseButton;
  33 import javafx.scene.paint.Color;
  34 import javafx.scene.robot.Robot;
  35 
  36 /**
  37  * iOS platform implementation class of test automation Robot.
  38  */
  39 final class IosRobot extends Robot {
  40 
  41     private long ptr = 0;
  42 
  43     // init and create native robot object
  44     private native long _init();
  45 
  46     @Override protected void create() {
  47         Application.checkEventThread();
  48         ptr = _init();
  49     }
  50 
  51     // release native robot object
  52     private native void _destroy(long ptr);
  53     @Override protected void destroy() {
  54         Application.checkEventThread();
  55         _destroy(ptr);
  56         ptr = 0;
  57     }
  58 
  59     // synthesize key press
  60     private native void _keyPress(long ptr, int code);
  61     @Override public void keyPress(KeyCode code) {
  62         Application.checkEventThread();
  63         if (ptr == 0) {
  64             return;
  65         }
  66         _keyPress(ptr, code.getCode());
  67     }
  68 
  69     // synthesize key release
  70     private native void _keyRelease(long ptr, int code);
  71     @Override public void keyRelease(KeyCode code) {
  72         Application.checkEventThread();
  73         if (ptr == 0) {
  74             return;
  75         }
  76         _keyRelease(ptr, code.getCode());
  77     }
  78 
  79     // synthesize mouse motion
  80     private native void _mouseMove(long ptr, int x, int y);
  81     @Override public void mouseMove(int x, int y) {
  82         Application.checkEventThread();
  83         if (ptr == 0) {
  84             return;
  85         }
  86         _mouseMove(ptr, x, y);
  87     }
  88 
  89     // synthesize mouse press of buttons
  90     private native void _mousePress(long ptr, int buttons);
  91     @Override public void mousePress(MouseButton button) {
  92         Application.checkEventThread();
  93         if (ptr == 0) {
  94             return;
  95         }
  96         _mousePress(ptr, GlassRobot.convertToRobotMouseButton(button));
  97     }
  98 
  99     @Override
 100     public void mousePress(MouseButton... buttons) {
 101         Application.checkEventThread();
 102         if (ptr == 0) {
 103             return;
 104         }
 105         _mousePress(ptr, GlassRobot.convertToRobotMouseButton(buttons));
 106     }
 107 
 108     // synthesize mouse release of buttons
 109     private native void _mouseRelease(long ptr, int buttons);
 110     @Override public void mouseRelease(MouseButton button) {
 111         Application.checkEventThread();
 112         if (ptr == 0) {
 113             return;
 114         }
 115         _mouseRelease(ptr, GlassRobot.convertToRobotMouseButton(button));
 116     }
 117 
 118     @Override
 119     public void mouseRelease(MouseButton... buttons) {
 120         Application.checkEventThread();
 121         if (ptr == 0) {
 122             return;
 123         }
 124         _mouseRelease(ptr, GlassRobot.convertToRobotMouseButton(buttons));
 125     }
 126 
 127     private native void _mouseWheel(long ptr, int wheelAmt);
 128     @Override protected void mouseWheel(int wheelAmt) {
 129         Application.checkEventThread();
 130         if (ptr == 0) {
 131             return;
 132         }
 133         _mouseWheel(ptr, wheelAmt);
 134     }
 135 
 136     // get x-coordinate of mouse location
 137     private native int _getMouseX(long ptr);
 138     @Override public int getMouseX() {
 139         Application.checkEventThread();
 140         if (ptr == 0) {
 141             return 0;
 142         }
 143         return _getMouseX(ptr);
 144     }
 145 
 146     // get x-coordinate of mouse location
 147     private native int _getMouseY(long ptr);
 148     @Override public int getMouseY() {
 149         Application.checkEventThread();
 150         if (ptr == 0) {
 151             return 0;
 152         }
 153         return _getMouseY(ptr);
 154     }
 155 
 156     private native int _getPixelColor(long ptr, int x, int y);
 157     @Override public Color getPixelColor(int x, int y) {
 158         Application.checkEventThread();
 159         if (ptr == 0) {
 160             return GlassRobot.convertFromIntArgb(0);
 161         }
 162         return GlassRobot.convertFromIntArgb(_getPixelColor(ptr, x, y));
 163     }
 164 
 165     @Override
 166     protected void getScreenCapture(int x, int y, int width, int height, int[] data) {
 167         Application.checkEventThread();
 168         if (ptr == 0) {
 169             return;
 170         }
 171         _getScreenCapture(ptr, x, y, width, height, data);
 172     }
 173 
 174     native private void _getScreenCapture(long ptr, int x, int y, int width, int height, int[] data);
 175 }
